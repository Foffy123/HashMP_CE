add_innovativeness_huge_effect = {
	if = {
		limit = { has_dlc = "Rule Britannia" }
		change_innovativeness = 5
	}
}

h_add_stability_or_adm_power_with_input = {
	custom_tooltip = $valuetooltip$

	hidden_effect = {
		set_variable = {
			which = stabilityCount
			value = 0
		}
	}

	while = {
		limit = {
			NOT = {
				check_variable = {
					which = stabilityCount
					value = $value$
				}
			}
		}

		if = {
			limit = {
				stability = 3
			}

			add_adm_power = 50
		}
		else = {
			add_stability = 1
		}

		change_variable = {
			which = stabilityCount
			value = 1
		}
	}
}

h_remove_stability_or_adm_power_with_input = {
	custom_tooltip = $valuetooltip$

	hidden_effect = {
		set_variable = {
			which = stabilityCount
			value = 0
		}
	}

	while = {
		limit = {
			NOT = {
				check_variable = {
					which = stabilityCount
					value = $value$
				}
			}
		}

		if = {
			limit = {
				stability = -3
			}

			add_adm_power = -50
		}
		else = {
			add_stability = -1
		}

		change_variable = {
			which = stabilityCount
			value = 1
		}
	}
}
