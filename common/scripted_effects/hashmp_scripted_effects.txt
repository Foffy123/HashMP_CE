add_innovativeness_huge_effect = {
	if = {
		limit = { has_dlc = "Rule Britannia" }
		change_innovativeness = 5
	}
}

h_add_mercantilism_or_dip_power_with_input = {
	custom_tooltip = $valuetooltip$

	hidden_effect = {
		set_variable = {
			which = mercantilismncount
			value = 0
		}
	}

	while = {
		limit = {
			NOT = {
				check_variable = {
					which = mercantilismncount
					value = $value$
				}
			}
		}

		if = {
			limit = {
				mercantilism = 100
			}

			add_dip_power = 50
		}
		else = {
			add_mercantilism = 1
		}

		change_variable = {
			which = mercantilismncount
			value = 1
		}
	}
}

h_recruit_marines_in_capital = {
	custom_tooltip = $recruittooltip$

	hidden_effect = {
		set_variable = {
			which = marinecount
			value = 0
		}
	}

	while = {	
		limit = {
			NOT = {
				check_variable = {
					which = marinecount
					value = $value$
				}
			}
		}

		capital_scope = {
			marine_infantry = $target$
		}

		change_variable = {
			which = marinecount
			value = 1
		}
	}
}
